#+BEGIN_SRC shell :exports results :results raw
  for f in *.tikz
  do
      filename=${f%.*}
      # Create pdf (if newer)
      [ $f -nt ./figures/${filename}.pdf ] && tikz2pdf --once $f > /dev/null 2>&1
      mkdir -p figures
      (
          cd figures
          mv ../$filename.pdf .
      )
      echo "* $filename"
      echo ":PROPERTIES:"
      echo ":HTML_CONTAINER_CLASS: tikz"
      echo ":CUSTOM_ID: hidden"
      echo ":END:"
      echo "[[./figures/$filename.png]]"
      echo "#+HTML: <div class=\"icons\">"
      typeset -A dict
      dict=( pdf fa-file-pdf-o tikz fa-file-text-o png fa-file-image-o )
      for k in "${(@k)dict}"
      do
          echo "[[./figures/$filename.$k][@@html:<i title=\"download $filename.$k\" class=\"fa ${dict[$k]} fa-lg\"></i>@@]]\n"
      done
      echo "#+HTML: </div>"
  done
#+END_SRC
